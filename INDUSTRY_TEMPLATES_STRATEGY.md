    id: generateId(),
    type: blockDef.type,
    content: populateBlockContent(blockDef, data),
    style: getBlockPreset(blockDef.preset),
  }));
  
  // Create microsite
  return await api.post('/microsites/from-template', {
    templateId,
    data,
    layout,
  });
}
